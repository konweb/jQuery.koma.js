jQuery.koma.js Demo

Default

$('.hoge').koma();

Option FPS

FPS 15

$('.hoge').koma( {fps: 15} );

FPS 30

$('.hoge').koma( {fps: 30} );

Stop/Restart Controls

Stop Event Handling 'koma_stop'

$('.js-koma').on('koma_stop', function(e, params){
  console.log('koma_stop');
  console.log(params);// params.$el
});

Restart Event Handling 'koma_restart'

$('.js-koma').on('koma_restart', function(e, params){
  console.log('koma_restart');
  console.log(params);// params.$el
});
